from math import isqrt

def czy_pierwsza(n: int)-> bool:
    if n < 2:
        return False
    for i in range(2, isqrt(n)+1):
        if n%i == 0:
            return False
    return True

def czy_polpierwsza(n: int)->bool:
    for i in range(2, isqrt(n)+1):
        if n%i == 0 and czy_pierwsza(i) and czy_pierwsza(n//i):
            return True

        
with open('dane/63ciagi.txt', 'r') as file:
    liczby=[]
    for line in file:
        line = line.strip()
        liczby.append(int(line, 2))

    ct, mini, maxi = 0, max(liczby),0
    for i in liczby:
        if czy_polpierwsza(i):
            ct+=1
            mini = min(mini, i)
            maxi = max(maxi, i)

print(ct, mini, maxi)
